\section*{Problem 2}
\begin{solution}
    Let \(S[i,j]\) be the length of the longest palindrome in substring \(S\)
    starting with the \(i\)th letter and ending with \(j\)th letter. We have
    \begin{align*}
        S[i,i] &= 1
    \end{align*}
    The time complexity of the algorithm is \(O(n^2)\).
\end{solution}
